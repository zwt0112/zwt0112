#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int main()
{
	FILE * fp;
	char buf[64] ={0};

	fp = fopen("foo.txt","r");
	if(NULL == fp)
	{
		perror("fopen");
		exit(1);
	}
	while(fread(buf, 1, 32, fp))
	{
		printf("%s", buf);
		memset(buf, 0, strlen(buf));
	}
	fclose(fp);
	return 0;
}
